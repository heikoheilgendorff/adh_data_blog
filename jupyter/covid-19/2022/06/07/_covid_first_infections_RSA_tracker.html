gdf = gpd.read_file('zip://'+data_path+'MDB_Wards_2016-shp.zip')

gdf = gdf.drop(columns=['OBJECTID', 'ProvinceCo','WardNumber','WardID','Year','Shape__Are', 'Shape__Len','DistrictMu','LocalMunic'])
gdf = gdf.rename(columns={'ProvinceNa':'Province'})

gdf_2 = gdf[['Province','geometry']]
gdf_2 = gdf_2.dissolve(by='Province')
gdf_2 = gdf_2.simplify(0.01,preserve_topology=False)
gdf_4 = gdf.simplify(0.01,preserve_topology=True)
gdf['geometry'] = gdf_4

def capitalize(gdf,col_name):
    names = gdf[col_name].to_list()
    names = [x.upper() for x in names]
    gdf[col_name] = names
    return gdf
gdf = capitalize(gdf,'District_1')
gdf = capitalize(gdf,'Province')
gdf = capitalize(gdf,'LocalMun_1')

Date_Name = 'Sub_date'

dates = df[Date_Name].drop_duplicates()
dates = dates.to_frame()
dates = pd.to_datetime(dates[Date_Name])
dates = dates.to_frame()
dates = dates.set_index(Date_Name, drop=True)
dates = dates.sort_index()
dates = dates.reset_index()
dates = dates[Date_Name].to_list()


def timeplot(x):
    date = dates[x]#.strftime('%x')
    print(date)
    
    date_data = df[df[Date_Name].isin(dates[0:x])]
    mergetemp=gdf.merge(date_data,how= 'left',left_on='LocalMun_1',right_on='Subdistrict')
    mergetemp.case = mergetemp.case.fillna(0)
    #mergetemp=world.merge(date_data,how= 'left',left_on='District_1',right_on='District')
    ax = gdf_2.plot(color='none',edgecolor='black')
    mergetemp.plot(ax=ax,
                   column='case',
                  #cmap='cool',
                  #edgecolor='black',
                  figsize=(25,20), 
                  legend = False,
                  legend_kwds={'label': "Areas with Confirmed COVID-19 cases",'orientation': "horizontal"})
    ax.axis('off')
